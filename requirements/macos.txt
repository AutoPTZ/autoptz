# AutoPTZ — macOS extras (Apple Silicon, M-series)
#
# Install after base:
#   pip install -r requirements/base.txt -r requirements/macos.txt
#
# The CoreML EP (ANE + GPU) is already bundled in the standard onnxruntime
# wheel on macOS arm64 — no separate package is needed.
# Hardware decode uses VideoToolbox via OpenCV/FFmpeg (system-provided).
#
# Verify CoreML EP is available:
#   python -c "import onnxruntime; print(onnxruntime.get_available_providers())"
#   # Should include 'CoreMLExecutionProvider'

# ── camera enumeration + native capture ────────────────────────────
# AVFoundation bindings: friendly device names + stable uniqueID, Continuity
# Camera detection (engine/discovery/usb.py:enumerate_cameras), AND native
# capture bound by uniqueID (engine/pipeline/avf_capture.py) — bypasses OpenCV's
# divergent AVFoundation device ordering so the picked camera is the one that
# opens. Optional — capture degrades to the cv2.VideoCapture path if absent, and
# enumeration falls back to generic "Camera N" entries.
pyobjc-framework-AVFoundation>=10.0
# Quartz pulls in CoreMedia (CMSampleBuffer) + CoreVideo (CVPixelBuffer) needed
# to pull BGRA pixels out of the capture sample buffers. (objc + Foundation ship
# with pyobjc-core / pyobjc-framework-Cocoa, transitive deps of the above.)
pyobjc-framework-Quartz>=10.0
# libdispatch: the GCD queue (dispatch_queue_create) the AVCaptureVideoDataOutput
# delivers sample buffers on. Shipped separately from pyobjc-core and the single
# easiest one to miss. NOW OPTIONAL: avf_capture._make_dispatch_queue falls back
# to calling dispatch_queue_create in libSystem via ctypes (wrapped with
# objc.objc_object), so native capture works with just the frameworks above even
# when this package is absent. Installing it is still preferred (uses the native
# binding directly).
pyobjc-framework-libdispatch>=10.0
